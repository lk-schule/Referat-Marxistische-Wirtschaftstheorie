  \frametitle{Wert einer Ware}

  \pause
  \begin{itemize}
    \item Gebrauchswert
    \pause
    \begin{itemize}
      \item Die Nützlichkeit der Ware bestimmt den Gebrauchswert
    \end{itemize}
    \pause
    \item Tauschwert

    \begin{itemize}
      \item Der Aufwand zur Produktion bestimmt den Tauschwert

  \end{itemize}
\end{frame}
